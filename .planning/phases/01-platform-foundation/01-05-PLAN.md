<objective>
Fix middleware route protection gap: replace dead `pathname.startsWith('/(app)')` condition with an inversion-based approach that protects ALL app routes by default.

Purpose: The current middleware checks `pathname.startsWith('/(app)')` which never matches because Next.js route groups are transparent to URLs. This means `/billing` (and any future route under `(app)/`) is not middleware-protected. The fix uses a public-route allowlist instead of a protected-route blocklist, so every new route is automatically protected.

Output: Updated `apps/web/middleware.ts` with correct, future-proof route protection.
</objective>

import { type NextRequest, NextResponse } from 'next/server'
import { updateSession } from '@/lib/supabase/middleware'

export async function middleware(request: NextRequest) {
  const { supabaseResponse, user } = await updateSession(request)
  const pathname = request.nextUrl.pathname

  // BUG: pathname.startsWith('/(app)') never matches — route groups are URL-transparent
  if (
    !user &&
    (pathname.startsWith('/(app)') || pathname.startsWith('/dashboard'))
  ) {
    return NextResponse.redirect(new URL('/auth', request.url))
  }

  if (user && pathname === '/auth') {
    return NextResponse.redirect(new URL('/dashboard', request.url))
  }

  return supabaseResponse
}

export const config = {
  matcher: [
    '/((?!_next/static|_next/image|favicon.ico|api/webhooks).*)',
  ],
}
