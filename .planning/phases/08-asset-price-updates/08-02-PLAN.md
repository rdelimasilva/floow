```typescript
// Source: brapi.dev/docs
// Requires env var: BRAPI_TOKEN (Bearer token from brapi.dev Startup plan)
// IMPORTANT: Use regularMarketTime (not new Date()) as price_date to handle market closures correctly

const BRAPI_TOKEN = process.env.BRAPI_TOKEN!
const BRAPI_BASE = 'https://brapi.dev/api'

export interface BrapiPrice {
  ticker: string
  priceCents: number
  priceDate: Date  // from regularMarketTime — actual trading date, not cron execution date
}

export async function fetchBrapiPrices(tickers: string[]): Promise<BrapiPrice[]> {
  if (tickers.length === 0) return []

  const url = `${BRAPI_BASE}/quote/${tickers.join(',')}?token=${BRAPI_TOKEN}&fundamental=false`
  const res = await fetch(url)

  if (!res.ok) {
    throw new Error(`[brapi] Error: ${res.status} ${await res.text()}`)
  }

  const data = await res.json() as {
    results: Array<{
      symbol: string
      regularMarketPrice: number | null
      regularMarketTime: string
    }>
  }

  return data.results
    .filter(r => r.regularMarketPrice != null)
    .map(r => ({
      ticker: r.symbol,
      priceCents: Math.round(parseFloat(r.regularMarketPrice!.toFixed(2)) * 100),
      priceDate: new Date(r.regularMarketTime),
    }))
}
```

**apps/web/lib/prices/coingecko-client.ts:**
```typescript
// Source: docs.coingecko.com/reference/simple-price
// Requires env var: COINGECKO_API_KEY (optional on Demo tier, recommended)
// Batches in chunks of 50 to avoid URL length limits (Demo: 30 req/min, 10k/month)

const COINGECKO_API_KEY = process.env.COINGECKO_API_KEY
const COINGECKO_BASE = 'https://api.coingecko.com/api/v3'
const CHUNK_SIZE = 50

export async function fetchCoinGeckoPrices(
  coinIds: string[],
  idToTicker: Record<string, string>,  // coingecko_id → asset ticker
): Promise<Array<{ ticker: string; priceCents: number; priceDate: Date }>> {
  if (coinIds.length === 0) return []

  const results: Array<{ ticker: string; priceCents: number; priceDate: Date }> = []

  for (let i = 0; i < coinIds.length; i += CHUNK_SIZE) {
    const chunk = coinIds.slice(i, i + CHUNK_SIZE)
    const url = `${COINGECKO_BASE}/simple/price?ids=${chunk.join(',')}&vs_currencies=brl&include_last_updated_at=true`

    const headers: Record<string, string> = { 'Content-Type': 'application/json' }
    if (COINGECKO_API_KEY) headers['x-cg-demo-api-key'] = COINGECKO_API_KEY

    const res = await fetch(url, { headers })
    if (!res.ok) throw new Error(`[coingecko] Error: ${res.status} ${await res.text()}`)

    const data = await res.json() as Record<string, { brl: number; last_updated_at: number }>

    for (const [coinId, prices] of Object.entries(data)) {
      const ticker = idToTicker[coinId]
      if (!ticker || prices.brl == null) {
        console.warn(`[coingecko] No ticker mapping or BRL price for coinId=${coinId}`)
        continue
      }
      results.push({
        ticker,
        priceCents: Math.round(parseFloat(prices.brl.toFixed(2)) * 100),
        priceDate: new Date(prices.last_updated_at * 1000),
      })
    }
  }

  return results
}
```

**apps/web/lib/prices/bcb-client.ts:**
```typescript
// Source: api.bcb.gov.br — series 12 (CDI), 11 (SELIC), 433 (IPCA)
// No auth required. Free, no documented rate limit.
// Date format in response: DD/MM/YYYY — must parse manually (not ISO 8601)
// Rate encoding: round(valor_percent * 1_000_000)
//   CDI 0.054266%/day → 54266
//   IPCA 0.70% monthly → 700000

const BCB_BASE = 'https://api.bcb.gov.br/dados/serie/bcdata.sgs'

async function fetchLatestRate(seriesId: number): Promise<{ data: string; valor: string }> {
  const res = await fetch(`${BCB_BASE}.${seriesId}/dados/ultimos/1?formato=json`)
  if (!res.ok) throw new Error(`[bcb] Series ${seriesId} error: ${res.status}`)
  const [row] = await res.json() as Array<{ data: string; valor: string }>
  if (!row) throw new Error(`[bcb] Series ${seriesId} returned empty array`)
  return row
}

export interface BcbIndicators {
  cdiDailyRateBps: number      // CDI daily rate — stored in economic_indicators
  selicDailyRateBps: number    // SELIC daily rate — stored in economic_indicators
  ipcaMonthlyRateBps: number   // IPCA monthly rate — stored in economic_indicators
  effectiveDate: Date           // Parsed from BCB DD/MM/YYYY date string
}

export async function fetchBcbIndicators(): Promise<BcbIndicators> {
  const [cdi, selic, ipca] = await Promise.all([
    fetchLatestRate(12),   // CDI daily
    fetchLatestRate(11),   // SELIC daily
    fetchLatestRate(433),  // IPCA monthly
  ])

  // BCB date format is DD/MM/YYYY
  const [day, month, year] = cdi.data.split('/').map(Number)
  const effectiveDate = new Date(year, month - 1, day)

  return {
    cdiDailyRateBps:     Math.round(parseFloat(cdi.valor) * 1_000_000),
    selicDailyRateBps:   Math.round(parseFloat(selic.valor) * 1_000_000),
    ipcaMonthlyRateBps:  Math.round(parseFloat(ipca.valor) * 1_000_000),
    effectiveDate,
  }
}
```

**apps/web/lib/prices/update-daily.ts** — orchestrator:
```typescript
import { getDb, globalAssetPrices, economicIndicators, assets } from '@floow/db'
import { eq, inArray, sql } from 'drizzle-orm'
import { fetchBrapiPrices } from './brapi-client'
import { fetchCoinGeckoPrices } from './coingecko-client'
import { fetchBcbIndicators } from './bcb-client'
import { recomputeOrgPositionSnapshots } from '@/lib/investments/position-snapshots'
import { invalidateTag, pricesTag, investmentsTag, snapshotsTag } from '@/lib/cache-tags'

export interface DailyPriceUpdateResult {
  b3Count: number
  cryptoCount: number
  indicatorsUpdated: boolean
  orgsRecomputed: number
}

export async function runDailyPriceUpdate(): Promise<DailyPriceUpdateResult> {
  const db = getDb()

  // 1. Collect ALL unique market-quoted assets across ALL orgs (no orgId filter — global)
  const marketAssets = await db
    .select({
      ticker: assets.ticker,
      coingeckoId: assets.coingeckoId,
      assetClass: assets.assetClass,
    })
    .from(assets)
    .where(eq(assets.pricingType, 'market_quoted'))

  // 2. Partition: B3 tickers vs crypto ids
  const b3Tickers = [...new Set(
    marketAssets
      .filter(a => a.assetClass !== 'crypto')
      .map(a => a.ticker)
  )]

  const cryptoAssets = marketAssets.filter(a => a.assetClass === 'crypto' && a.coingeckoId != null)
  const cryptoIds = [...new Set(cryptoAssets.map(a => a.coingeckoId!))]
  const idToTicker: Record<string, string> = {}
  for (const a of cryptoAssets) {
    if (a.coingeckoId) idToTicker[a.coingeckoId] = a.ticker
  }

  // Warn about crypto assets missing coingecko_id
  const missingIds = marketAssets.filter(a => a.assetClass === 'crypto' && !a.coingeckoId)
  if (missingIds.length > 0) {
    console.warn(`[price-update] ${missingIds.length} crypto asset(s) have null coingecko_id — skipping: ${missingIds.map(a => a.ticker).join(', ')}`)
  }

  // 3. Fetch all three sources in parallel
  const [b3Prices, cryptoPrices, indicators] = await Promise.all([
    fetchBrapiPrices(b3Tickers),
    fetchCoinGeckoPrices(cryptoIds, idToTicker),
    fetchBcbIndicators(),
  ])

  // 4. Upsert market prices to global_asset_prices
  const allPrices = [...b3Prices, ...cryptoPrices]
  if (allPrices.length > 0) {
    await db
      .insert(globalAssetPrices)
      .values(allPrices.map(p => ({
        ticker: p.ticker,
        priceDate: p.priceDate,
        priceCents: p.priceCents,
        source: b3Prices.includes(p as typeof b3Prices[number]) ? 'brapi' : 'coingecko',
      })))
      .onConflictDoUpdate({
        target: [globalAssetPrices.ticker, globalAssetPrices.priceDate],
        set: {
          priceCents: sql`EXCLUDED.price_cents`,
          updatedAt: new Date(),
        },
      })
  }

  // 5. Upsert BCB indicators to economic_indicators (NOT global_asset_prices — separate table by design)
  await db
    .insert(economicIndicators)
    .values([
      { indicator: 'CDI',  refDate: indicators.effectiveDate, rateBps: indicators.cdiDailyRateBps,    source: 'bcb' },
      { indicator: 'SELIC', refDate: indicators.effectiveDate, rateBps: indicators.selicDailyRateBps,  source: 'bcb' },
      { indicator: 'IPCA', refDate: indicators.effectiveDate, rateBps: indicators.ipcaMonthlyRateBps, source: 'bcb' },
    ])
    .onConflictDoUpdate({
      target: [economicIndicators.indicator, economicIndicators.refDate],
      set: { rateBps: sql`EXCLUDED.rate_bps`, updatedAt: new Date() },
    })

  // 6. Find affected orgs and recompute their position snapshots
  const updatedTickers = allPrices.map(p => p.ticker)
  let orgsRecomputed = 0

  if (updatedTickers.length > 0) {
    const affectedOrgs = await db
      .selectDistinct({ orgId: assets.orgId })
      .from(assets)
      .where(inArray(assets.ticker, updatedTickers))

    for (const { orgId } of affectedOrgs) {
      await recomputeOrgPositionSnapshots(orgId)
      invalidateTag(pricesTag(orgId))
      invalidateTag(investmentsTag(orgId))
      invalidateTag(snapshotsTag(orgId))
      orgsRecomputed++
    }
  }

  return {
    b3Count: b3Prices.length,
    cryptoCount: cryptoPrices.length,
    indicatorsUpdated: true,
    orgsRecomputed,
  }
}
```

NOTE: The source detection in step 4 above is a simplification. A cleaner approach is to add `source: 'brapi'` to each b3Price entry and `source: 'coingecko'` to each cryptoPrice entry inside the client functions, or add a source tag when merging. Adjust accordingly if the typing becomes awkward — the important constraint is that the upsert works correctly.

**apps/web/app/api/prices/update-daily/route.ts** — API route (follow cfo/run-daily pattern exactly):
```typescript
import { NextResponse } from 'next/server'
import { runDailyPriceUpdate } from '@/lib/prices/update-daily'

const SERVICE_ROLE_KEY = process.env.SUPABASE_SERVICE_ROLE_KEY!

export async function POST(request: Request) {
  const authHeader = request.headers.get('authorization')
  if (authHeader !== `Bearer ${SERVICE_ROLE_KEY}`) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })
  }

  try {
    const result = await runDailyPriceUpdate()
    return NextResponse.json({ ok: true, ...result })
  } catch (err) {
    console.error('[price-update] Daily run failed:', err)
    return NextResponse.json({ error: 'Price update failed' }, { status: 500 })
  }
}
```

```typescript
import type { Config } from '@netlify/functions'

export default async () => {
  const siteUrl = process.env.URL || process.env.DEPLOY_PRIME_URL
  if (!siteUrl) {
    console.error('[price-update] No URL env var — cannot call update-daily endpoint')
    return new Response('No URL configured', { status: 500 })
  }

  const serviceRoleKey = process.env.SUPABASE_SERVICE_ROLE_KEY
  if (!serviceRoleKey) {
    console.error('[price-update] No SUPABASE_SERVICE_ROLE_KEY')
    return new Response('No service role key', { status: 500 })
  }

  try {
    const response = await fetch(`${siteUrl}/api/prices/update-daily`, {
      method: 'POST',
      headers: {
        Authorization: `Bearer ${serviceRoleKey}`,
        'Content-Type': 'application/json',
      },
    })

    const body = await response.text()
    console.log(`[price-update] Status: ${response.status}, Body: ${body}`)

    return new Response(body, { status: response.status })
  } catch (err) {
    console.error('[price-update] Failed:', err)
    return new Response(String(err), { status: 500 })
  }
}

export const config: Config = {
  schedule: '0 19 * * 1-5',  // 19:00 UTC weekdays — 90-min buffer after B3 close at 17:30 BRT
}
```
