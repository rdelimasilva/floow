```typescript
describe('computeAccrualPrice', () => {
  // CDI at 0.054266%/day = rateBps 54266
  const CDI_BPS = 54266

  it('returns purchase price unchanged when referenceDate equals purchaseDate', () => {
    const today = new Date(2026, 0, 15)  // Jan 15, 2026
    expect(computeAccrualPrice(1_000_000, CDI_BPS, 1.0, today, today)).toBe(1_000_000)
  })

  it('returns purchase price unchanged when businessDays <= 0', () => {
    const purchaseDate = new Date(2026, 0, 15)
    const refDate = new Date(2026, 0, 14)  // day before purchase
    expect(computeAccrualPrice(1_000_000, CDI_BPS, 1.0, purchaseDate, refDate)).toBe(1_000_000)
  })

  it('accrues upward over 365 calendar days at CDI rate', () => {
    const purchaseDate = new Date(2025, 0, 15)  // Jan 15, 2025
    const refDate = new Date(2026, 0, 15)        // Jan 15, 2026 (365 days)
    const result = computeAccrualPrice(1_000_000, CDI_BPS, 1.0, purchaseDate, refDate)
    // At ~0.054266%/day over 252 business days: ~14.88% gain → ~R$10,148.80
    expect(result).toBeGreaterThan(1_000_000)
    expect(result).toBeLessThan(1_200_000)  // sanity upper bound
  })

  it('90% CDI spread produces less accrual than 100% CDI', () => {
    const purchaseDate = new Date(2025, 0, 15)
    const refDate = new Date(2026, 0, 15)
    const full = computeAccrualPrice(1_000_000, CDI_BPS, 1.0, purchaseDate, refDate)
    const reduced = computeAccrualPrice(1_000_000, CDI_BPS, 0.9, purchaseDate, refDate)
```

```typescript
/**
 * Computes current accrual-based asset value using the Brazilian 252-business-day convention.
 *
 * DO NOT use estimateAssetValue() from asset-valuation.ts for CDI-linked products.
 * That function uses calendar-day compounding (daysElapsed/365) which produces
 * systematically incorrect valuations for Brazilian fixed income.
 *
 * @param purchasePriceCents - Original purchase price in integer cents
 * @param dailyRateBps - Daily rate in basis points * 10000 (from economic_indicators.rate_bps)
 *   e.g., CDI at 0.054266%/day is stored as 54266
 * @param spread - Multiplier for rate (1.0 = 100% CDI, 0.9 = 90% CDI)
 * @param purchaseDate - Date of purchase
 * @param referenceDate - Valuation date (defaults to today)
 * @returns Current value in integer cents (Math.round applied)
 */
export function computeAccrualPrice(
  purchasePriceCents: number,
  dailyRateBps: number,
  spread: number,
  purchaseDate: Date,
  referenceDate: Date = new Date(),
): number {
  const msPerDay = 86_400_000
  const calendarDays = Math.floor(
    (referenceDate.getTime() - purchaseDate.getTime()) / msPerDay
  )
  // Approximate business days using 252/365 ratio
  const businessDays = Math.round(calendarDays * 252 / 365)
  if (businessDays <= 0) return purchasePriceCents

  // dailyRateBps: 54266 means 0.054266% per day (stored as percent * 1_000_000)
  // Step 1: 54266 / 1_000_000 = 0.054266 (still a percentage)
  // Step 2: divide by 100 to get decimal fraction: 0.054266 / 100 = 0.00054266
  const dailyRatePercent = (dailyRateBps / 1_000_000) * spread
  const dailyRate = dailyRatePercent / 100

  return Math.round(purchasePriceCents * Math.pow(1 + dailyRate, businessDays))
}
```
