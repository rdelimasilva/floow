```typescript
export async function getLatestPrices(orgId: string): Promise<Record<string, number>> {
  return unstable_cache(
    async () => {
      const db = getDb()

      // Fetch all assets to know their pricingType
      const orgAssets = await db
        .select({
          id: assets.id,
          ticker: assets.ticker,
          pricingType: assets.pricingType,
        })
        .from(assets)
        .where(eq(assets.orgId, orgId))

      const marketQuotedTickers = orgAssets
        .filter(a => a.pricingType === 'market_quoted')
        .map(a => a.ticker)

      const marketQuotedAssetIds = orgAssets
        .filter(a => a.pricingType === 'market_quoted')
        .map(a => a.id)

      // Fetch global prices for market_quoted tickers (one per unique ticker, latest date)
      const globalPriceRows = marketQuotedTickers.length > 0
        ? await db.execute<{ ticker: string; price_cents: number }>(
            sql`SELECT DISTINCT ON (ticker) ticker, price_cents
                FROM global_asset_prices
                WHERE ticker = ANY(${marketQuotedTickers})
                ORDER BY ticker, price_date DESC`
          )
        : []

      const globalByTicker: Record<string, number> = {}
      for (const row of globalPriceRows) {
        globalByTicker[row.ticker] = row.price_cents
      }

      // Fetch per-org manual prices as fallback (existing asset_prices table)
      const manualPriceRows = await db.execute<{ asset_id: string; price_cents: number }>(
        sql`SELECT DISTINCT ON (asset_id) asset_id, price_cents
            FROM asset_prices
            WHERE org_id = ${orgId}
            ORDER BY asset_id, price_date DESC`
      )

      const manualByAssetId: Record<string, number> = {}
      for (const row of manualPriceRows) {
        manualByAssetId[row.asset_id] = row.price_cents
      }

      // Build result: global price preferred for market_quoted; manual fallback for all
      const latestPrices: Record<string, number> = {}
      for (const asset of orgAssets) {
        if (asset.pricingType === 'market_quoted' && globalByTicker[asset.ticker] != null) {
          latestPrices[asset.id] = globalByTicker[asset.ticker]
        } else if (manualByAssetId[asset.id] != null) {
          latestPrices[asset.id] = manualByAssetId[asset.id]
        }
        // accrual_based assets without manual price: omitted (snapshot recomputation handles them)
      }

      return latestPrices
    },
    ['investment-latest-prices', orgId],
    { tags: [pricesTag(orgId)], revalidate: 300 },
  )()
}
```

```typescript
async function getLatestPriceCents(db: DbClient, orgId: string, assetId: string): Promise<number> {
  // Check asset's pricingType to determine source
  const [assetRow] = await db
    .select({ ticker: assets.ticker, pricingType: assets.pricingType })
    .from(assets)
    .where(and(eq(assets.orgId, orgId), eq(assets.id, assetId)))
    .limit(1)

  if (!assetRow) return 0

  if (assetRow.pricingType === 'market_quoted') {
    // Prefer global price for market_quoted assets
    const [globalPrice] = await db
      .select({ priceCents: globalAssetPrices.priceCents })
      .from(globalAssetPrices)
      .where(eq(globalAssetPrices.ticker, assetRow.ticker))
      .orderBy(desc(globalAssetPrices.priceDate))
      .limit(1)

    if (globalPrice) return globalPrice.priceCents
  }

  // Fallback: per-org manual asset_prices (also used for accrual_based with manual override)
  const [latestPrice] = await db
    .select({ priceCents: assetPrices.priceCents })
    .from(assetPrices)
    .where(and(eq(assetPrices.orgId, orgId), eq(assetPrices.assetId, assetId)))
    .orderBy(desc(assetPrices.priceDate))
    .limit(1)

  return latestPrice?.priceCents ?? 0
}
```
